Following are the sql queries for milestone 3:

1. CREATE STREAM KSQL_AUTH (
       SVC STRUCT<CASE_NUMBER VARCHAR, SVC_ID VARCHAR, SVC_FAC_ID VARCHAR, SVC_TYPE VARCHAR, SVC_CODE VARCHAR, SVC_FAC_NAME VARCHAR, SVC_PHY_NAME VARCHAR, SVC_PHY_ID VARCHAR, "N/A" VARCHAR>,
       PAT STRUCT<CASE_NUMBER VARCHAR, PAT_FIRST_NAME VARCHAR, PAT_ID VARCHAR, PAT_MIDDLE_NAME VARCHAR, PAT_LAST_NAME VARCHAR, PAT_SEX VARCHAR, PAT_DOB VARCHAR, PAT_PLANE_TYPE VARCHAR, PAT_PLAN_NAME VARCHAR, "N/A" VARCHAR>,
       CAS STRUCT<CASE_NUMBER VARCHAR, CASE_CODE VARCHAR, CASE_TYPE VARCHAR, CASE_AUTH_TYPE VARCHAR, CASE_END_DATE VARCHAR, CASE_START_DATE VARCHAR, "N/A" VARCHAR>,
       SUB STRUCT<MEM_FIRST_NAME VARCHAR, MEM_CITY VARCHAR, MEM_PIN VARCHAR, MEM_ADD_1 VARCHAR, MEM_ID VARCHAR, MEM_MIDDLE_NAME VARCHAR, MEM_LAST_NAME VARCHAR, MEM_ADD_2 VARCHAR, CASE_NUMBER VARCHAR, "N/A" VARCHAR>)
   WITH (VALUE_FORMAT='JSON', KAFKA_TOPIC='Auth-Topic');

2. CREATE STREAM KSQL_SUB WITH (KAFKA_TOPIC='Sub-Topic', VALUE_FORMAT='JSON') AS SELECT
   SUB,'KSQL' AS IDENTIFIER FROM KSQL_AUTH
   WHERE SUB IS NOT NULL;

3. CREATE STREAM KSQL_CAS WITH (KAFKA_TOPIC='Case-Topic', VALUE_FORMAT='JSON') AS SELECT
   CAS,'KSQL' AS IDENTIFIER FROM KSQL_AUTH
   WHERE CAS IS NOT NULL;

4. CREATE STREAM KSQL_PAT WITH (KAFKA_TOPIC='Pat-Topic', VALUE_FORMAT='JSON') AS SELECT
   PAT,'KSQL' AS IDENTIFIER FROM KSQL_AUTH
   WHERE PAT IS NOT NULL;

5. CREATE STREAM KSQL_SVC WITH (KAFKA_TOPIC='Ser-Topic', VALUE_FORMAT='JSON') AS SELECT
   SVC,'KSQL' AS IDENTIFIER FROM KSQL_AUTH
   WHERE SVC IS NOT NULL;
